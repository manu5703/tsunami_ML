# Training queries — Forest Covertype (Roosevelt National Forest, Colorado)
# Target zone: HIGH ALPINE (Elevation > 3200, Slope > 15)  — HighAlpine zone
# Covers ~12% of the dataset. Tsunami builds fine-grained partitions across this region,
# making it especially precise for the tighter Krummholz subset used as nearby queries.
# Columns: Elevation, Aspect, Slope, Horiz_Dist_Hydrology, Vert_Dist_Hydrology,
#          Horiz_Dist_Roadways, Hillshade_9am, Hillshade_Noon, Hillshade_3pm,
#          Horiz_Dist_Fire_Points

# Core alpine zone
SELECT COUNT(*) FROM data WHERE Elevation > 3200 AND Slope > 15
SELECT COUNT(*) FROM data WHERE Elevation > 3300 AND Slope > 18
SELECT COUNT(*) FROM data WHERE Elevation > 3200 AND Hillshade_9am < 210

# Alpine + roadways
SELECT COUNT(*) FROM data WHERE Elevation > 3200 AND Horiz_Dist_Roadways > 1500
SELECT COUNT(*) FROM data WHERE Elevation > 3300 AND Horiz_Dist_Roadways > 2000

# Slope + hillshade
SELECT COUNT(*) FROM data WHERE Slope > 15 AND Hillshade_9am < 210
SELECT COUNT(*) FROM data WHERE Slope > 18 AND Hillshade_Noon < 220

# Alpine + hydrology
SELECT COUNT(*) FROM data WHERE Elevation > 3200 AND Horiz_Dist_Hydrology > 300
SELECT COUNT(*) FROM data WHERE Elevation > 3300 AND Vert_Dist_Hydrology > 40

# 3-attribute
SELECT COUNT(*) FROM data WHERE Elevation > 3200 AND Slope > 15 AND Hillshade_9am < 210
SELECT COUNT(*) FROM data WHERE Elevation > 3300 AND Slope > 18 AND Horiz_Dist_Roadways > 1500
SELECT COUNT(*) FROM data WHERE Elevation > 3200 AND Hillshade_9am < 210 AND Horiz_Dist_Roadways > 1500
SELECT COUNT(*) FROM data WHERE Elevation > 3200 AND Slope > 15 AND Horiz_Dist_Hydrology > 300

# 4-attribute
SELECT COUNT(*) FROM data WHERE Elevation > 3200 AND Slope > 15 AND Hillshade_9am < 210 AND Horiz_Dist_Roadways > 1500
SELECT COUNT(*) FROM data WHERE Elevation > 3300 AND Slope > 18 AND Hillshade_Noon < 220 AND Horiz_Dist_Roadways > 2000
SELECT COUNT(*) FROM data WHERE Elevation > 3200 AND Slope > 15 AND Horiz_Dist_Hydrology > 300 AND Horiz_Dist_Roadways > 1500

# Aggregations
SELECT AVG(Elevation) FROM data WHERE Slope > 15 AND Hillshade_9am < 210
SELECT AVG(Slope) FROM data WHERE Elevation > 3200 AND Horiz_Dist_Roadways > 1500
SELECT AVG(Hillshade_9am) FROM data WHERE Elevation > 3200 AND Slope > 15
SELECT SUM(Horiz_Dist_Roadways) FROM data WHERE Elevation > 3200 AND Slope > 15
SELECT SUM(Horiz_Dist_Hydrology) FROM data WHERE Elevation > 3200 AND Slope > 15 AND Hillshade_9am < 210
